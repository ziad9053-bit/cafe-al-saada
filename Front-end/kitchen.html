async function loadOrders() { const container = document.getElementById('orders-container'); if (!container) return; // تأكد أن الصفحة تحتوي على الحاوية المطلوبة const { data, error } = await window.supabase .from('orders') .select('*') .eq('status', 'pending') .order('created_at', { ascending: false }); if (error) { console.error("خطأ في جلب البيانات:", error); return; } // التحقق إذا كانت البيانات فارغة if (!data || data.length === 0) { container.innerHTML = '

لا توجد طلبات معلقة حالياً.

'; return; } // بناء الواجهة container.innerHTML = data.map(order => `

طاولة رقم: ${order.table_number || 'غير معروف'}

    ${Array.isArray(order.items) ? order.items.map(item => `
  • ${item.name} × ${item.quantity}
    • `).join('') : '
  • لا توجد أصناف
    • '}
`).join(''); } async function markAsDone(orderId) { const { error } = await window.supabase .from('orders') .update({ status: 'completed' }) .eq('id', orderId); if (error) { console.error("خطأ أثناء التحديث:", error); } else { loadOrders(); // تحديث القائمة فوراً بعد الضغط } } // تشغيل الوظيفة عند تحميل الصفحة loadOrders(); // تحديث تلقائي كل 10 ثوانٍ setInterval(loadOrders, 10000);
